By default imlp_ptr uses the std::allocator for memory allocation.

It can be changed with 

struct Book : boost::impl_ptr<Book, my_allocator>::shared { ... };
struct Book : boost::impl_ptr<Book, my_allocator>::unique { ... };
struct Book : boost::impl_ptr<Book, my_allocator>::copied { ... };
