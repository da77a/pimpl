// In header: <detail/inplace.hpp>

template<size_t s, size_t a = std::size_t(-1)> 
struct storage {
  // member classes/structs/unions
  template<typename impl_type> 
  struct type : private always_storage::template type< impl_type > {
    // types
    typedef typename always_storage< s, a >::template type< impl_type > base;       
    typedef typename base::traits_type                                  traits_type;
    typedef typename base::traits_ptr                                   traits_ptr; 

    // public member functions
    traits_ptr get_traits() const;
    void set_traits(const traits_ptr);
    template<typename derived_type, typename... arg_types> 
      void emplace(arg_types &&...);
  };

  // public data members
  static size_t constexpr size;
  static size_t constexpr alignment;
  static bool constexpr has_null_state;
};
